Preview 업데이트 방법 `cmd + option + p`

```swift

// View는 여기서 Class가 아닌 Protocol

protocol View {
	associatedtype Body: View
	var body: Self.Body get { }
}

// View 프로토콜을 재귀호출 -> Text Image Color 와 같이 실제 컨텐츠를 표현하는 기본 뷰(Primitive Views)와 Stack, Group, GeometryReader 같은 컨테이뷰에는 더는 재귀호출이 일어나지 않도록 Never 타입이 활용됌

extension Text: View {
	typealias Body = Never
}

// body라는 계산(연산) 속성을 가짐
struct ContentView: View {
	var body: some View {
		Text("Hello World")
			.font(.largeTitle)
			.foregroundColor(.blue)
	}
}
```

- SwiftUI에서는 font, foregroundColor와 같은 메서드를 수식어(modifier)라고 칭함
